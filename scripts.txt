# this script looks at which rating is used the most
cat reviews.Fifty-Shades.csv | cut -f 2 | sort | uniq -c | sort -nr

# this script looks at what rating the most helpfull reviews got.
cat reviews.Fifty-Shades.csv | cut -f 2,3 | sort -g -k 2

the results from the script above are a big mess so with awk I filtered out the reviews which got at least 200 votes as a helpful review
cat reviews.Fifty-Shades.csv | cut -f 2,3 | sort -g -k 2 | awk -F '\t' '{if($2 >= 200) print}'

# This script takes the rating with the year and month
$ cat reviews.Fifty-Shades.csv | cut -f 1,2 | sed -E 's/(-[0-9]+)-[0-9]+/\1/'

# This script shows us which ratings were given in which month of a year and how often these ratings were given
cat reviews.Fifty-Shades.csv | cut -f 1,2 | sed -E 's/(-[0-9]+)-[0-9]+/\1/' | sort | uniq -c

# this script shows us how many reviews were given in a month
cat reviews.Fifty-Shades.csv | cut -f 1 | sed -E 's/(-[0-9]+)-[0-9]+/\1/' | sort | uniq -c

# this script shows us in which reviews certain words are used, for example 'bought'

# this script selects the reviews that rated the book with a 4 or a 5
$ cat reviews.Fifty-Shades.csv | awk -F '\t' '{if ($2 >= 2.5) print}' | cut -f 5

# this script selects the reviews that rated the book with a 1 or a 2
$ cat reviews.Fifty-Shades.csv | awk -F '\t' '{if ($2 <= 2.5) print}' | cut -f 5
